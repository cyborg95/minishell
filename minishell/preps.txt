### Notes About This Tree

Executor Folder

This is your pipex evolution.

execute_pipe.c is recursive, like your current pipex.

external_exec.c is your execute_cmd() upgraded.

Builtins Folder

Only implement logic here.

Do not fork for builtins that must run in the parent (cd, export, unset, exit).

Execution folder handles when to fork.

Env Folder

Convert envp into linked list (t_env) at shell startup.

All env/export/unset operations go through these helpers.

Signals Folder

One global int g_signal only.

Handles Ctrl-C, Ctrl-, Ctrl-D behavior.

Utils Folder

Norm-friendly helpers.

Freeing, string manipulation, error messages.

char	*get_env_value(t_env *env, char *key)
{
	while (env)
	{
		if (ft_strcmp(env->key, key) == 0)
			return (env->value);
		env = env->next;
	}
	return (NULL);
}


char	*expand_var(char *arg, t_shell *shell)
{
	char	*value;

	if (arg[0] != '$')
		return (ft_strdup(arg));
	value = get_env_value(shell->env, arg + 1);
	if (!value)
		return (ft_strdup(""));
	return (ft_strdup(value));
}

void	expand_args(t_node *node, t_shell *shell)
{
	int		i;
	char	*tmp;

	if (!node || !node->argv)
		return ;
	i = 0;
	while (node->args[i])
	{
		tmp = expand_var(node->argv[i], shell);
		free(node->argv[i]);
		node->argv[i] = tmp;
		i++;
	}
}

void	expand_tree(t_node *node, t_shell *shell)
{
	if (!node)
		return ;
	expand_argv(node, shell);
	expand_tree(node->left, shell);
	expand_tree(node->right, shell);
}
